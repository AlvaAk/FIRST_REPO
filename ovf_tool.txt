#!/bin/bash
# This script deploys the connector on the ESXi machine 1.1.72.94.
# Asks for the password, which is : Fpass
ESXI="1.1.72.89"
USER=aalva-18.04-con9th
TS=$(date -u +%d_%b_%H_%M)
PASS=$(cat ./esxi-pass)

ovftool \
    --X:logFile=ovftool-log.txt  \
    --X:logLevel=verbose  \
    --noSSLVerify \
    --name="${USER}-${TS}" \
    --network="aalva-port-group1" \
    --powerOn \
    ./ovf_directory/Connector.ovf "vi://root:${PASS}@${ESXI}"
